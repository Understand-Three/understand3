# 启动 IPFS 并保持后台运行
```bash
ipfs daemon & 
```
# 添加当前目录下的所有内容
```bash
 ipfs add ./* -w  
```

# 查看已连接的 peer node
```bash
ipfs swarm peers
```


```bash
ipfs stats bitswap
```

# webui
```bash
ipfs config --json API.HTTPHeaders.Access-Control-Allow-Methods '["POST","GET"]'
> ipfs config --json API.HTTPHeaders.Access-Control-Allow-Origin '["http://localhost:3000", "https://webui.ipfs.io", "http://127.0.0.1:5001"]'
> ipfs config --json API.HTTPHeaders.Access-Control-Allow-Methods '["POST"]'
```
